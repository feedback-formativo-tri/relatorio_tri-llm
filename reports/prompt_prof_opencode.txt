  const btns = document.querySelectorAll('.tab-btn');

}

<div class="card">
<h3>Como ler este painel</h3>
<p class="muted" style="margin:0;">Use os gráficos para identificar (1) diferenças entre UFs, (2) onde o desempenho observado foge do esperado pelo modelo, e (3) quais itens/alunos priorizar. As imagens podem ser ampliadas com um clique.</p>
